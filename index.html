/* docs (with defaults) */
export default ({
  el, // parent element
  item = 'li', // selector of items
  loop = false, // infinite carousel
  autoplay = false, // moves items on every raf
  pausable = true, // dont stop autoplay on hover
  snap = false, // makes items move in jumps
  align, // undefined || 'center' -> centers items when snap is enabled
  lerp: lerpAmount = [0.1, 0.1], // easing of scroll [mobile, desktop]
  onChange = () => {}, // callback when a new item becomes 'active'
  onUpdate = () => {}, // callback on every raf
})

/* callbacks are half implemented ðŸ™ƒ */

Examples:

/* Default */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.default'),
})
    
/* Looped */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.looped'),
  loop: true
})
    
/* Autoplay */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.autoplay'),
  loop: true,
  autoplay: true
})
    
/* Speed */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.speed'),
  loop: true,
  autoplay: 5,
})
    
/* Non-pausable */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.no-pause'),
  loop: true,
  autoplay: true,
  pausable: false
})
    
/* Snap */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.snap'),
  loop: true,
  autoplay: 5,
  pausable: false,
  snap: true
})
    
/* Center */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.center'),
  loop: true,
  autoplay: 5,
  pausable: false,
  snap: true,
  align: 'center'
})
    
/* Callback */
import dragScroll from './drag-scroll'

dragScroll({
  el: document.querySelector('.callback'),
  loop: true,
  autoplay: true,
  onUpdate: (state) => {
    const output = document.querySelector('output')

    output.textContent = state.current
  }
})
